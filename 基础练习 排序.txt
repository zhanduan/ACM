const int N=10000;
bool compare(int a,int b)
{
      return a<b;   //升序排列，如果改为return a>b，则为降序

}
int main()
{	
	int i,n,a[N];
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	sort(a,a+n,compare);
	for(i=0;i<n;i++)
		cout<<a[i]<<" ";
	return 0;
}
