//算法训练 输出米字形
#include<iostream>
#include<cstring> 
using namespace std;

int main()
{
	char a[100][100];
	int n,k,i,j;
	cin>>n;
	memset(a,'.',sizeof(a));
	for(k=n-1;k>0;k--)
		for(i=n-k-1;i<2*n-1-(n-k-1);i+=k)
			for(j=n-k-1;j<2*n-1-(n-k-1);j+=k)
			{
				a[i][j]='A'+(n-k-1);
				//cout<<a[i][j];
			}
	a[n-1][n-1]='A'+n-1;
	for(i=0;i<2*n-1;i++)
	{
		for(j=0;j<2*n-1;j++)
		{
			cout<<a[i][j];
		}
		cout<<endl;
	}
	return 0;
}
