//算法提高 交换Easy
#include <iostream>
using namespace std;
#define N 1000
int main()
{
	int n,m,id1,id2,k;
	int a[N];
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>a[i];
    for(int i=0;i<m;i++)
	{
		cin>>id1>>id2;
		id1--;id2--;
		k=a[id1];
		a[id1]=a[id2];
		a[id2]=k;
	}
	for(int i=0;i<n;i++)
		cout<<a[i]<<endl;
    return 0;
}
