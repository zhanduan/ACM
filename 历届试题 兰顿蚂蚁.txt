//历届试题 兰顿蚂蚁
#include <iostream>
using namespace std;
#define N 100
int main()
{
	//ifstream cin("in.txt");
//	ofstream cout("out.txt");
	int m,n,i,j,k,x,y,dir;
	int a[N][N];
	char d;
	cin>>m>>n;
	for(i=0;i<m;i++)
		for(j=0;j<n;j++)
			cin>>a[i][j];
	cin>>x>>y>>d>>k;
	int dif[4][2]={{-1,0},{0,1},{1,0},{0,-1}};
	if(d=='U')dir=0;
	else if(d=='R')dir=1;
	else if(d=='D')dir=2;
	else if(d=='L')dir=3;
	while(k--)
	{
		//cout<<x<<" "<<y<<" "<<k<<endl;
		if(a[x][y])//黑格，右转90度
		{
			a[x][y]=0;
			dir=(dir+1)%4;
			x=x+dif[dir][0];
			y=y+dif[dir][1];		
		}
		else
		{
			a[x][y]=1;
			dir=(dir+3)%4;
			x=x+dif[dir][0];
			y=y+dif[dir][1];
		}
		
	}
	cout<<x<<" "<<y; 
    return 0;
}
